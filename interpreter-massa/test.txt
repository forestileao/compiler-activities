int num1;
int num2;
int resultado;
float valor;
boolean condicao;
char letra;

num1 = 10;
num2 = 5;
valor = 3.14;
letra = 'X';
condicao = true;

write("--- Valores iniciais ---");
write("num1 = ");
write(num1);
write("num2 = ");
write(num2);
write("valor = ");
write(valor);
write("letra = ");
write(letra);

resultado = num1 + num2;
write("Soma: ");
write(resultado);

resultado = num1 - num2;
write("Subtração: ");
write(resultado);

resultado = num1 * num2;
write("Multiplicação: ");
write(resultado);

resultado = num1 / num2;
write("Divisão: ");
write(resultado);

write("--- Testando IF-ELSE ---");

if (num1 > num2) then {
    write("Condição verdadeira: num1 é maior que num2");
    resultado = num1;
} else {
    write("Condição falsa: num1 não é maior que num2");
    resultado = num2;
}

if (num1 < num2) then {
    write("Condição verdadeira: num1 é menor que num2");
    resultado = num1;
} else {
    write("Condição falsa: num1 não é menor que num2");
    resultado = num2;
}

if (condicao) then {
    write("Condição booleana verdadeira");
} else {
    write("Condição booleana falsa");
}

if ((num1 > num2) AND (valor > 3.0)) then {
    write("Ambas condições são verdadeiras");
} else {
    write("Pelo menos uma condição é falsa");
}

if ((num1 < num2) OR (valor > 3.0)) then {
    write("Pelo menos uma condição é verdadeira");
} else {
    write("Ambas condições são falsas");
}

write("--- Testando conversões de tipo ---");

valor = num1;
write("Inteiro para float:");
write(valor);

num1 = 65;
letra = num1;
write("Inteiro para char (ASCII 65 = A):");
write(letra);

write("--- Entrada do usuário ---");
write("Digite um número inteiro:");
read(num1);
write("Você digitou:");
write(num1);

if (num1 > 10) then {
    write("O número é maior que 10");
} else {
    write("O número é menor ou igual a 10");
}
